                            {

                // Fallback to basic responses if API fails

        async function sendMessage() {
            const message = chatInput.value.trim();
            if (!message) return;
            
            // Add user message
            addMessage(message, true);
            chatInput.value = '';
            
            // Show typing indicator
            const typingDiv = addMessage('', false, true);
            
            try {
                // Get response from ChatGPT (or fallback)
                const response = await getChatGPTResponse(message);
                
                // Remove typing indicator
                typingDiv.remove();
                
                // Add AI response
                addMessage(response, false);
                
                // Add follow-up suggestion after certain responses
                setTimeout(() => {
                    if (response.includes('personalized quote') || response.includes('Curtis Ray') || response.includes('(386) 400-5886')) {
                        addMessage("üí° Tip: You can also scroll down to fill out the quote form for a quick response!", false);
                    }
                }, 1500);
                
            } catch (error) {
                console.error('Error getting AI response:', error);
                
                // Remove typing indicator
                typingDiv.remove();
                
                // Show error message
                addMessage("I'm having trouble connecting right now. Please call Curtis Ray directly at (386) 400-5886 for immediate assistance with your insurance questions!", false);
            }
        }
